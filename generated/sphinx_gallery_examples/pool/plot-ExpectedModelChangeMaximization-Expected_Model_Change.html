Expected Model Change#

  9 import numpy as np
 10 from matplotlib import pyplot as plt, animation
 11 from scipy.stats import uniform
 12
 13 from skactiveml.utils import MISSING_LABEL, labeled_indices, is_labeled
 14
 15 from skactiveml.regressor import NICKernelRegressor
 16 from skactiveml.pool import ExpectedModelChangeMaximization
 17
 18 random_state = np.random.RandomState(0)
 19
 20
 21 def true_function(X_):
 22     return (X_**3 + 2 * X_**2 + X_ - 1).flatten()
 23
 24 n_samples = 100
 25 X = np.concatenate(
 26         [uniform.rvs(0, 1.5, 9 * n_samples // 10, random_state=random_state),
 27          uniform.rvs(1.5, 0.5, n_samples // 10, random_state=random_state)]
 28     ).reshape(-1, 1)
 29
 30 noise = np.vectorize(lambda x : random_state.rand() * 1.5 if x < 1
 31                                 else random_state.rand() * 0.5)
 32
 33 # Build a dataset.
 34 y_true = true_function(X) + noise(X).flatten()
 35 y = np.full(shape=y_true.shape, fill_value=MISSING_LABEL)
 36 X_test = np.linspace(0, 2, num=100).reshape(-1, 1)
 37
 38 # Initialise the classifier.
 39 reg = NICKernelRegressor(random_state=random_state, metric_dict={'gamma': 15.0})
 40 # Initialise the query strategy.
 41 qs = ExpectedModelChangeMaximization()
 42
 43
 44 # Preparation for plotting.
 45 fig, (ax_1, ax_2) = plt.subplots(2, 1, sharex=True)
 46 artists = []
 47
 48 # The active learning cycle:
 49 n_cycles = 20
 50 for c in range(n_cycles):
 51     # Fit the classifier.
 52     reg.fit(X, y)
 53
 54     # Get labeled instances.
 55     X_labeled = X[labeled_indices(y)]
 56
 57     # Query the next instance/s.
 58     query_idx = qs.query(X=X, y=y, reg=reg)
 59
 60     # Plot the labeled data.
 61     coll_old = list(ax_1.collections) + list(ax_2.collections)
 62     title = ax_1.text(
 63         0.5,
 64         1.05,
 65         f"Prediction after acquring {c} labels",
 66         size=plt.rcParams["axes.titlesize"],
 67         ha="center",
 68         transform=ax_1.transAxes,
 69     )
 70
 71     _, utilities_test = qs.query(
 72         X=X, y=y, reg=reg, candidates=X_test, return_utilities=True
 73     )
 74
 75     utilities_test = (utilities_test - utilities_test.min()).flatten()
 76     if np.any(utilities_test != utilities_test[0]):
 77         utilities_test /= utilities_test.max()
 78
 79     is_lbld = is_labeled(y)
 80
 81     (utility_line,) = ax_2.plot(X_test, utilities_test, c="green")
 82     utility_fill = plt.fill_between(
 83         X_test.flatten(), utilities_test, color="green", alpha=0.3
 84     )
 85
 86     ax_1.scatter(X[~is_lbld], y_true[~is_lbld], c="lightblue")
 87     ax_1.scatter(X[is_lbld], y[is_lbld], c="orange")
 88
 89     y_pred = reg.predict(X_test)
 90     (prediction_line,) = ax_1.plot(X_test, y_pred, c="black")
 91
 92     coll_new = list(ax_1.collections) + list(ax_2.collections)
 93     coll_new.append(title)
 94     artists.append(
 95         [x for x in coll_new if (x not in coll_old)]
 96         + [utility_line, utility_fill, prediction_line]
 97     )
 98
 99     # Label the queried instances.
100     y[query_idx] = y_true[query_idx]
101
102 ani = animation.ArtistAnimation(fig, artists, interval=1000, blit=True)
